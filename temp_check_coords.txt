using System;
using WorldZones.WorldGen;

var generator = new WorldGenerator("HHcLC5acQt");
var type = generator.GetType();

var offset0 = (double)type.GetField("offset0", System.Reflection.BindingFlags.NonPublic | System.Reflection.BindingFlags.Instance).GetValue(generator);
var offset1 = (double)type.GetField("offset1", System.Reflection.BindingFlags.NonPublic | System.Reflection.BindingFlags.Instance).GetValue(generator);
var offset2 = (double)type.GetField("offset2", System.Reflection.BindingFlags.NonPublic | System.Reflection.BindingFlags.Instance).GetValue(generator);

// Test a location from ground truth
int worldX = -11709;
int worldZ = 11853;

Console.WriteLine($"Testing coord ({worldX}, {worldZ}):");
Console.WriteLine();

// Forest calculation
double forestX = (offset2 + worldX) * 0.001;
double forestZ = (offset2 + worldZ) * 0.001;
Console.WriteLine($"BlackForest input: ({forestX:F3}, {forestZ:F3})");
Console.WriteLine($"After modulo 256: ({forestX % 256:F3}, {forestZ % 256:F3})");

var biome = generator.GetBiome(worldX, worldZ);
Console.WriteLine($"Predicted biome: {biome}");
Console.WriteLine($"Ground truth: BlackForest");
